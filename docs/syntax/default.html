Default



Default are containers for storing default values.

shell 
{
    default
    {
        // view = auto, compact, small, medium, large, wide
        view = view.compact
        
        // theme = auto, dark, light, white, black
        theme = theme.black
        
        // text = [normal, hover, disabled]
        text = [#ffffff, default, #808080]
        
        // block = [normal, hover, radius]
        block = [#000, #444, 5]

        // separator = [color, size]
        separator = [#ff00ff, 1]
        
        // frame = [border color, border size]
        frame = [#000, 3]
        
        // font = [size, font name]
        font = [12, 'Brush Script']
       
        // Chevron, Check Mark, Radio Bullet
        symbol = #ff0000
       
        // Enable/Disable drop shadow
        shadow = false
    }
}
